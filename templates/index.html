Setup Checklist
Before running scripts in ServiceNow PDI
1. Deploy CEJO online and get your public URL
2. In ServiceNow PDI: System Web Services → Scripted REST APIs → New
3. Create a Resource, paste the script, set HTTP Method
4. Replace cejo-ipam.onrender.com with your actual CEJO URL
5. Test using header X-CEJO-API-Key: cejo-api-2024
1 — Allocate IP from ServiceNow
Scripted REST API · POST
(function process(request, response) {
  var CEJO = 'https://cejo-ipam.onrender.com';
  var KEY  = 'cejo-api-2024';
  var body = request.body.data;
  var rm = new sn_ws.RESTMessageV2();
  rm.setEndpoint(CEJO + '/api/v1/allocate_ip');
  rm.setHttpMethod('POST');
  rm.setRequestHeader('Content-Type', 'application/json');
  rm.setRequestHeader('X-CEJO-API-Key', KEY);
  rm.setRequestBody(JSON.stringify({
    network_cidr: body.network_cidr || '10.0.10.0/24',
    device_name:  body.device_name,
    device_type:  body.device_type || 'Server',
    owner:        body.owner || gs.getUserName()
  }));
  var r    = rm.execute();
  var code = r.getStatusCode();
  var data = JSON.parse(r.getBody());
  if (code === 201) {
    var ci = new GlideRecord('cmdb_ci_ip_address');
    ci.newRecord();
    ci.setValue('ip_address',  data.ip);
    ci.setValue('mac_address', data.mac);
    ci.setValue('name',        data.device);
    ci.insert();
  }
  response.setStatus(code);
  response.setBody(data);
})(request, response);
2 — Release IP from ServiceNow
Scripted REST API · POST
(function process(request, response) {
  var CEJO = 'https://cejo-ipam.onrender.com';
  var KEY  = 'cejo-api-2024';
  var ip   = request.body.data.ip_address;
  var rm = new sn_ws.RESTMessageV2();
  rm.setEndpoint(CEJO + '/api/v1/release_ip');
  rm.setHttpMethod('POST');
  rm.setRequestHeader('Content-Type', 'application/json');
  rm.setRequestHeader('X-CEJO-API-Key', KEY);
  rm.setRequestBody(JSON.stringify({ ip_address: ip }));
  var r    = rm.execute();
  var data = JSON.parse(r.getBody());
  if (r.getStatusCode() === 200) {
    var ci = new GlideRecord('cmdb_ci_ip_address');
    ci.addQuery('ip_address', ip);
    ci.query();
    if (ci.next()) { ci.setValue('install_status', 7); ci.update(); }
  }
  response.setStatus(r.getStatusCode());
  response.setBody(data);
})(request, response);

  -H "X-CEJO-API-Key: cejo-api-2024" \
  -H "Content-Type: application/json" \
  -d '{"network_cidr":"10.0.10.0/24","name":"test-srv","device_type":"Server"}' \
  https://cejo-ipam.onrender.com/api/v1/add_device

  https://cejo-ipam.onrender.com/api/v1/release_ip
