📌 Setup Checklist
Prerequisites before running ServiceNow scripts
1. Deploy CEJO online (Render/Railway/Fly.io) and get public URL
2. In your ServiceNow PDI: System Web Services → Scripted REST APIs → New
3. Create a Resource with HTTP Method POST/GET and paste the script below
4. Replace YOUR-APP.onrender.com with your actual CEJO URL
5. Test with Try It or Postman Collection
1 — Allocate IP from ServiceNow Workflow
Scripted REST API Resource · POST · /api/cejo/v1/allocate
(function process(request, response) {
  var CEJO = 'https://YOUR-APP.onrender.com';
  var KEY  = 'cejo-api-2024';

  var body        = request.body.data;
  var networkCidr = body.network_cidr || '10.0.10.0/24';
  var deviceName  = body.device_name  || 'sn-device-' + new GlideDateTime().getNumericValue();
  var deviceType  = body.device_type  || 'Server';
  var owner       = body.owner        || gs.getUserName();
  var desc        = body.description  || 'Allocated via ServiceNow';

  var rm = new sn_ws.RESTMessageV2();
  rm.setEndpoint(CEJO + '/api/v1/allocate_ip');
  rm.setHttpMethod('POST');
  rm.setRequestHeader('Content-Type', 'application/json');
  rm.setRequestHeader('X-CEJO-API-Key', KEY);
  rm.setRequestBody(JSON.stringify({
    network_cidr: networkCidr,
    device_name:  deviceName,
    device_type:  deviceType,
    owner:        owner,
    description:  desc
  }));

  var r    = rm.execute();
  var code = r.getStatusCode();
  var data = JSON.parse(r.getBody());

  if (code === 201) {
    // Write to CMDB - IP Address CI
    var ci = new GlideRecord('cmdb_ci_ip_address');
    ci.newRecord();
    ci.setValue('ip_address',  data.ip);
    ci.setValue('mac_address', data.mac);
    ci.setValue('name',        data.device);
    ci.setValue('short_description', 'CEJO auto-allocated: ' + data.network);
    ci.insert();

    gs.log('CEJO: Allocated ' + data.ip + ' to ' + data.device, 'CEJO');
    response.setStatus(201);
    response.setBody(data);
  } else {
    response.setStatus(code);
    response.setBody({ error: data.error });
  }
})(request, response);
2 — Release IP from ServiceNow
Triggered when a CI is decommissioned
(function process(request, response) {
  var CEJO = 'https://YOUR-APP.onrender.com';
  var KEY  = 'cejo-api-2024';

  var ip = request.body.data.ip_address;
  if (!ip) {
    response.setStatus(400);
    response.setBody({ error: 'ip_address required' });
    return;
  }

  var rm = new sn_ws.RESTMessageV2();
  rm.setEndpoint(CEJO + '/api/v1/release_ip');
  rm.setHttpMethod('POST');
  rm.setRequestHeader('Content-Type', 'application/json');
  rm.setRequestHeader('X-CEJO-API-Key', KEY);
  rm.setRequestBody(JSON.stringify({ ip_address: ip }));

  var r    = rm.execute();
  var data = JSON.parse(r.getBody());

  // Retire CMDB CI
  if (r.getStatusCode() === 200) {
    var ci = new GlideRecord('cmdb_ci_ip_address');
    ci.addQuery('ip_address', ip);
    ci.query();
    if (ci.next()) {
      ci.setValue('install_status', 7); // Retired
      ci.update();
    }
    gs.log('CEJO: Released ' + ip, 'CEJO');
  }

  response.setStatus(r.getStatusCode());
  response.setBody(data);
})(request, response);
3 — Business Rule: Auto-Allocate on CI Creation
Before Insert on cmdb_ci_server · Condition: ip_address.nil()
// Business Rule on table: cmdb_ci_server
// When: Before / Insert
// Condition: current.ip_address.nil()

(function executeRule(current, previous) {
  var CEJO = 'https://YOUR-APP.onrender.com';
  var KEY  = 'cejo-api-2024';

  var rm = new sn_ws.RESTMessageV2();
  rm.setEndpoint(CEJO + '/api/v1/add_device');
  rm.setHttpMethod('POST');
  rm.setRequestHeader('Content-Type', 'application/json');
  rm.setRequestHeader('X-CEJO-API-Key', KEY);
  rm.setRequestBody(JSON.stringify({
    network_cidr: '10.0.10.0/24',
    name:         current.getValue('name'),
    device_type:  'Server',
    owner:        gs.getUserName(),
    description:  'Auto-registered from CMDB: ' + current.getValue('short_description')
  }));

  var r    = rm.execute();
  var data = JSON.parse(r.getBody());

  if (r.getStatusCode() === 201) {
    current.setValue('ip_address',  data.ip_address);
    current.setValue('mac_address', data.mac_address);
  } else {
    gs.log('CEJO BR error: ' + data.error, 'CEJO');
  }
})(current, previous);

5 — curl Test Commands
Run these to verify CEJO is working from CLI or Postman
# Health (no auth)
curl https://YOUR-APP.onrender.com/health

# Full hierarchical status
curl -H "X-CEJO-API-Key: cejo-api-2024" \
  https://YOUR-APP.onrender.com/api/v1/status

# Create a Block
curl -X POST -H "X-CEJO-API-Key: cejo-api-2024" \
  -H "Content-Type: application/json" \
  -d '{"name":"Dev Cloud","cidr":"172.20.0.0/16","location":"AWS-eu-west-1"}' \
  https://YOUR-APP.onrender.com/api/v1/create_block

# Create a Network inside the block
curl -X POST -H "X-CEJO-API-Key: cejo-api-2024" \
  -H "Content-Type: application/json" \
  -d '{"block_cidr":"172.20.0.0/16","name":"k8s-nodes","cidr":"172.20.1.0/24","vlan_id":50}' \
  https://YOUR-APP.onrender.com/api/v1/create_network

# Add a device (auto-allocate IP+MAC)
curl -X POST -H "X-CEJO-API-Key: cejo-api-2024" \
  -H "Content-Type: application/json" \
  -d '{"network_cidr":"10.0.10.0/24","name":"k8s-master","device_type":"Server","owner":"devops"}' \
  https://YOUR-APP.onrender.com/api/v1/add_device

# Allocate specific IP

# Release an IP
